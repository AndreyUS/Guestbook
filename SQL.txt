CREATE TABLE `comments` (
`id` INT NOT NULL AUTO_INCREMENT,
`name` VARCHAR NOT NULL,
`email` VARCHAR NOT NULL,
`text` LONGTEXT NOT NULL,
`time` TIME NOT NULL,
`date` DATE NOT NULL,
PRIMARY KEY(`id`) 
)

CREATE TABLE `users` (
`id` INT NOT NULL AUTO_INCREMENT,
`login` VARCHAR(128) NOT NULL,
`password` VARCHAR(128) NOT NULL,
PRIMARY KEY(`id`)
)
